SELECT D.DEPT_ID, D.DEPT_NAME, 
COUNT(DECODE(GENDER,'MALE',1,NULL))"MALE",
COUNT(DECODE(GENDER,'FEMALE',1,NULL))"FEMALE",
COUNT(GENDER)"TOTAL"
FROM EMP_SRINI E, DEPT_SRINI D
WHERE D.DEPT_ID = E.DEPT_ID
GROUP BY D.DEPT_ID,D.DEPT_NAME;
