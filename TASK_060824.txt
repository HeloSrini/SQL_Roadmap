 WHERE CUST_ID='C1002';

3.
UPDATE SALES_INV_LN_SRINI
SET SIL_UNIT_PRICE = SIL_UNIT_PRICE * 5.5
WHERE SIL_INV_NO IN (SELECT SIH_INV_NO 
FROM SALES_INV_HDR_SRINI 
WHERE TO_CHAR(SIH_INV_DATE, 'MON') = 'FEB');

4.
UPDATE SALES_INV_HDR_SRINI
SET SIH_INV_DATE = SIH_INV_DATE + 74
WHERE SIH_INV_NO= '2324-002';

5.
UPDATE SALES_INV_LN_SRINI
SET SIL_UNIT_PRICE = SIL_UNIT_PRICE * 4
WHERE SIL_ITEM_ID IN (SELECT SIL_ITEM_ID FROM ITEM_MASTER_SRINI,CUST_MASTER_SRINI,SALES_INV_LN_SRINI,SALES_INV_HDR_SRINI
WHERE SIL_ITEM_ID = ITEM_ID AND CUST_ID=SIH_CUST_ID
AND SIH_INV_NO = SIL_INV_NO AND  CUST_ID= 'C1004' AND
ITEM_ID='FG1002');
