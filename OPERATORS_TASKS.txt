CREATE TABLE EMPS(
EMPID		NUMBER,
EMPNAME 	VARCHAR2(10),
SALARY 		NUMBER(7,2),
DA		NUMBER(7,2),
HRA		NUMBER(7,2),
CCA		NUMBER(7,2),
PF 		NUMBER(7,2),
PPF 		NUMBER(7,2),
CROSS_SALARY 	NUMBER(7,2),
TOTAL_DEDUCE	NUMBER(7,2),
NET_SALARY	NUMBER(7,2)
);


INSERT INTO EMPS(EMPID,EMPNAME,SALARY,DA,HRA,CCA,PF,PPF,CROSS_SALARY,TOTAL_DEDUCE,NET_SALARY)
VALUES ('&EMPID','&EMPNAME',SALARY,DA,HRA,CCA,PF,PPF,CROSS_SALARY,TOTAL_DEDUCE,NET_SALARY)
