- 1.SELECT EMP_NAME ,TRUNC(MONTHS_BETWEEN(SYSDATE,DOB)/12) AGE  FROM EMP_SRINI WHERE TRUNC(MONTHS_BETWEEN(SYSDATE,DOB)/12) BETWEEN 30 AND 45;

- 2.SELECT EMP_NAME,ROUND(MONTHS_BETWEEN(SYSDATE,DOJ)/12) AS SERVICE_PERIOD FROM EMP_SRINI WHERE (MONTHS_BETWEEN(SYSDATE,DOJ)/12) BETWEEN 10 AND 15;

- 3.SELECT D.DEPT_ID,D.DEPT_NAME,SUM(BASIC_SAL) AS SALARY FROM DEPT_SRINI D, EMP_SRINI E
    WHERE D.DEPT_ID=E.DEPT_ID GROUP BY D.DEPT_ID,D.DEPT_NAME
    HAVING SUM(BASIC_SAL)>100000;

- 4.SELECT EMP_NAME FROM EMP_SRINI WHERE  (MONTHS_BETWEEN(SYSDATE,DOB)/12)  >35 AND BASIC_SAL>35000 AND (MONTHS_BETWEEN(SYSDATE,DOJ)/12) <=10;

- 5.SELECT D.DEPT_ID,D.DEPT_NAME,COUNT(EMPID) FROM DEPT_SRINI D,EMP_SRINI E
     WHERE D.DEPT_ID=E.DEPT_ID GROUP BY D.DEPT_ID,D.DEPT_NAME 

- 6. SELECT A.DEPT_ID,A.DEPT_NAME,SUM(BASIC_SAL) FROM DEPT_SRINI A, EMP_SRINI  B
