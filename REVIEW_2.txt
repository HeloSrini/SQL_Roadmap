SELECT DEPT_NAME,AVG(BASIC_SAL)FROM DEPT_SRINI D,EMP_SRINI F
WHERE D.DEPT_ID = F.DEPT_ID GROUP BY DEPT_NAME

SELECT D.DEPT_ID,D.DEPT_NAME,SUM(BASIC_SAL) AS SALARY FROM DEPT_SRINI D, EMP_SRINI E
WHERE D.DEPT_ID=E.DEPT_ID GROUP BY D.DEPT_ID,D.DEPT_NAME
HAVING SUM(BASIC_SAL)>100000

SELECT D.DEPT_ID,D.DEPT_NAME,COUNT(EMP_ID)
      FROM DEPT_ARUN D,EMP_ARUN E
 WHERE D.DEPT_ID=E.DEPT_ID


SELECT D.DEPT_ID,D.DEPT_NAME,COUNT(EMPID) FROM DEPT_SRINI D,EMP_SRINI E
WHERE D.DEPT_ID=E.DEPT_ID GROUP BY DEPT_ID

SELECT D.DEPT_ID,D.DEPT_NAME,COUNT(EMPID) FROM DEPT_SRINI D,EMP_SRINI E WHERE D.DEPT_ID=E.DEPT_ID AND BASIC_SAL BETWEEN 125000 AND 165000 GROUP BY D.DEPT_ID,D.DEPT_NAME;
 

SELECT D.DEPT_ID,D.DEPT_NAME,ROUND(AVG(MONTHS_BETWEEN(SYSDATE,DOB) /12)) AGE FROM DEPT_SRINI D, EMP_SRINI E WHERE D.DEPT_ID=E.DEPT_ID GROUP BY D.DEPT_ID,D.DEPT_NAME;

SELECT D.DEPT_ID,D.DEPT_NAME, COUNT(DECODE(GENDER,'FEMALE',1,NULL)) FEMALE,COUNT(DECODE(GENDER,'MALE',1,NULL)) MALE, COUNT(GENDER) GENDER FROM DEPT_SRINI D, EMP_SRINI E
